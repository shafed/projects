\begin{equation*}
  \begin{aligned}
    Y_2 &= D_7 \lor D_6 \lor D_5 \lor D_4 \\
    Y_1 &= D_7 \lor D_6 \lor (\overline{D_5} \cdot \overline{D_4} \cdot
    D_3) \lor (\overline{D_5} \cdot \overline{D_4} \cdot D_2) \\
    Y_0 &= D_7 \lor (\overline{D_6} \cdot D_5) \lor (\overline{D_6}
    \cdot \overline{D_4} \cdot D_3) \lor (\overline{D_6} \cdot
    \overline{D_4} \cdot \overline{D_2} \cdot D_1) \\
  \end{aligned}
\end{equation*}
